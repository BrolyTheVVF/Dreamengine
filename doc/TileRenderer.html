checkout Renderer for standard parameters

 var haldShip = new DE.GameObject( {
  x: 500, y: 500,
  renderer: new DE.TileRenderer( {
    "imageName": "ship",
    "tilesizes": { "width": "100", "height": 200 },
    "tileposition": { "x": "100", "y": 0 }
  } )
} );
